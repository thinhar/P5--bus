% This is an example file showing how code snippets should be inserted into LaTeX documents.
% Todo entries are listed in order of apperance after the table of contents.

% todo is used for stuff that needs to be done later
\todo{stuff you want to do}
% Unsure is used for possible corrections
\unsure{stuff you are unsure about}
% Change is used for stuff that should definately be changed later
\change{stuff you want to change}
% Info is for general info
\info{stuff you want info about}
